## üìå Extending an interface 

```typescript 
//  –†–∞—Å—à–∏—Ä–µ–Ω–∏–µ –¥–ª—è interface

  name: string;
}

  honey: boolean;
}

  return {
    name: "Baloo",
    honey: true
  };
};

const bear = getBear();
bear.name;
bear.honey;
```
